231 void Adafruit_ILI9341::setRotation(uint8_t m) {
232 
233  uint8_t data;
234  rotation = m % 4; // can't be higher than 3
235  switch (rotation) {
236  case 0:
237  data = MADCTL_MX | MADCTL_BGR;
238  _width = ILI9341_TFTWIDTH;
239  _height = ILI9341_TFTHEIGHT;
240  break;
241  case 1:
242  data = MADCTL_MV | MADCTL_BGR;
243  _width = ILI9341_TFTHEIGHT;
244  _height = ILI9341_TFTWIDTH;
245  break;
246  case 2:
247  data = MADCTL_MY | MADCTL_BGR;
248  _width = ILI9341_TFTWIDTH;
249  _height = ILI9341_TFTHEIGHT;
250  break;
251  case 3:
252  data = MADCTL_MX | MADCTL_MY | MADCTL_MV | MADCTL_BGR;
253  _width = ILI9341_TFTHEIGHT;
254  _height = ILI9341_TFTWIDTH;
255  break;
256  }
257  transmitCmdData(ILI9341_MADCTL, &data, 1);
258 }
