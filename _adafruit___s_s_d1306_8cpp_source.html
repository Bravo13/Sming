105 // the most basic function, set a single pixel
106 void Adafruit_SSD1306::drawPixel(int16_t x, int16_t y, uint16_t color) {
107  if ((x < 0) || (x >= width()) || (y < 0) || (y >= height()))
108  return;
109 
110  // check rotation, move pixel around if necessary
111  switch (getRotation()) {
112  case 1:
113  swap(x, y);
114  x = WIDTH - x - 1;
115  break;
116  case 2:
117  x = WIDTH - x - 1;
118  y = HEIGHT - y - 1;
119  break;
120  case 3:
121  swap(x, y);
122  y = HEIGHT - y - 1;
123  break;
124  }
125 
126  // x is which column
127  switch (color)
128  {
129  case WHITE: buffer[x+ (y/8)*SSD1306_LCDWIDTH] |= (1 << (y&7)); break;
130  case BLACK: buffer[x+ (y/8)*SSD1306_LCDWIDTH] &= ~(1 << (y&7)); break;
131  case INVERSE: buffer[x+ (y/8)*SSD1306_LCDWIDTH] ^= (1 << (y&7)); break;
132  }
133 
134 }

161 void Adafruit_SSD1306::begin(uint8_t vccstate, uint8_t i2caddr, bool reset) {
162  _vccstate = vccstate;
163  _i2caddr = i2caddr;
164 
165  // set pin directions
166  if (sid != -1){
167  pinMode(dc, OUTPUT);
168  pinMode(cs, OUTPUT);
169  csport = portOutputRegister(digitalPinToPort(cs));
170  cspinmask = digitalPinToBitMask(cs);
171  dcport = portOutputRegister(digitalPinToPort(dc));
172  dcpinmask = digitalPinToBitMask(dc);
173  if (!hwSPI){
174  // set pins for software-SPI
175  pinMode(sid, OUTPUT);
176  pinMode(sclk, OUTPUT);
177  clkport = portOutputRegister(digitalPinToPort(sclk));
178  clkpinmask = digitalPinToBitMask(sclk);
179  mosiport = portOutputRegister(digitalPinToPort(sid));
180  mosipinmask = digitalPinToBitMask(sid);
181  }
182  if (hwSPI){
183  SPI.begin ();
184 #if defined(__SAM3X8E__)
185  SPI.setClockDivider (9); // 9.3 MHz
186 #elif defined(__ESP8266_EX__)
187  ;// SPI clock divider not implemented yet!
188 #else
189  SPI.setClockDivider (SPI_CLOCK_DIV2); // 8 MHz
190 #endif
191  }
192  }
193  else
194  {
195  // I2C Init
196  Wire.begin();
197 #ifdef __SAM3X8E__
198  // Force 400 KHz I2C, rawr! (Uses pins 20, 21 for SDA, SCL)
199  TWI1->TWI_CWGR = 0;

210  // bring reset low

216  // turn on VCC (9V?)
217  }

488  #if SSD1306_LCDHEIGHT == 32

492  ssd1306_command(1); // Page end address
493  #endif
494

496  {
497  // SPI
498  *csport |= cspinmask;
499  *dcport |= dcpinmask;
500  *csport &= ~cspinmask;
501 
502  for (uint16_t i=0; i<(SSD1306_LCDWIDTH*SSD1306_LCDHEIGHT/8); i++) {
503  fastSPIwrite(buffer[i]);
504  //ssd1306_data(buffer[i]);
505  }
506  *csport |= cspinmask;
507  }

592 void Adafruit_SSD1306::drawFastHLineInternal(int16_t x, int16_t y, int16_t w, uint16_t color) {
593  // Do bounds/limit checks
594  if(y < 0 || y >= HEIGHT) { return; }
595 
596  // make sure we don't try to draw below 0
597  if(x < 0) {
598  w += x;
599  x = 0;
600  }
601 
602  // make sure we don't go off the edge of the display
603  if( (x + w) > WIDTH) {
604  w = (WIDTH - x);
605  }
606 
607  // if our width is now negative, punt
608  if(w <= 0) { return; }
609 
610  // set up the pointer for movement through the buffer
611  register uint8_t *pBuf = buffer;
612  // adjust the buffer pointer for the current row
613  pBuf += ((y/8) * SSD1306_LCDWIDTH);
614  // and offset x columns in
615  pBuf += x;
616 
617  register uint8_t mask = 1 << (y&7);
618 
619  switch (color)
620  {
621  case WHITE: while(w--) { *pBuf++ |= mask; }; break;
622  case BLACK: mask = ~mask; while(w--) { *pBuf++ &= mask; }; break;
623  case INVERSE: while(w--) { *pBuf++ ^= mask; }; break;

661 void Adafruit_SSD1306::drawFastVLineInternal(int16_t x, int16_t __y, int16_t __h, uint16_t color) {
662 
663  // do nothing if we're off the left or right side of the screen
664  if(x < 0 || x >= WIDTH) { return; }
665 
666  // make sure we don't try to draw below 0
667  if(__y < 0) {
668  // __y is negative, this will subtract enough from __h to account for __y being 0
669  __h += __y;
670  __y = 0;
671 
672  }
673 
674  // make sure we don't go past the height of the display
675  if( (__y + __h) > HEIGHT) {
676  __h = (HEIGHT - __y);
677  }
678 
679  // if our height is now negative, punt
680  if(__h <= 0) {
681  return;
682  }
683 
684  // this display doesn't need ints for coordinates, use local byte registers for faster juggling
685  register uint8_t y = __y;
686  register uint8_t h = __h;
687 
688 
689  // set up the pointer for fast movement through the buffer
690  register uint8_t *pBuf = buffer;
691  // adjust the buffer pointer for the current row
692  pBuf += ((y/8) * SSD1306_LCDWIDTH);
693  // and offset x columns in
694  pBuf += x;
695 
696  // do the first partial byte, if necessary - this requires some masking
697  register uint8_t mod = (y&7);
698  if(mod) {
699  // mask off the high n bits we want to set
700  mod = 8-mod;
701 
702  // note - lookup table results in a nearly 10% performance improvement in fill* functions
703  // register uint8_t mask = ~(0xFF >> (mod));
704  static uint8_t premask[8] = {0x00, 0x80, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC, 0xFE };
705  register uint8_t mask = premask[mod];
706 
707  // adjust the mask if we're not going to reach the end of this byte
708  if( h < mod) {
709  mask &= (0XFF >> (mod-h));
710  }
711 
712  switch (color)
713  {
714  case WHITE: *pBuf |= mask; break;
715  case BLACK: *pBuf &= ~mask; break;
716  case INVERSE: *pBuf ^= mask; break;
717  }
718 
719  // fast exit if we're done here!
720  if(h<mod) { return; }
721 
722  h -= mod;
723 
724  pBuf += SSD1306_LCDWIDTH;
725  }
726 
727 
728  // write solid bytes while we can - effectively doing 8 rows at a time
729  if(h >= 8) {
730  if (color == INVERSE) { // separate copy of the code so we don't impact performance of the black/white write version with an extra comparison per loop
731  do {
732  *pBuf=~(*pBuf);
733 
734  // adjust the buffer forward 8 rows worth of data
735  pBuf += SSD1306_LCDWIDTH;
736 
737  // adjust h & y (there's got to be a faster way for me to do this, but this should still help a fair bit for now)
738  h -= 8;
739  } while(h >= 8);
740  }
741  else {
742  // store a local value to work with
743  register uint8_t val = (color == WHITE) ? 255 : 0;
744 
745  do {
746  // write our value in
747  *pBuf = val;
748 
749  // adjust the buffer forward 8 rows worth of data
750  pBuf += SSD1306_LCDWIDTH;
751 
752  // adjust h & y (there's got to be a faster way for me to do this, but this should still help a fair bit for now)
753  h -= 8;
754  } while(h >= 8);
755  }
756  }
757 
758  // now do the final partial byte, if necessary
759  if(h) {
760  mod = h & 7;
761  // this time we want to mask the low bits of the byte, vs the high bits we did above
762  // register uint8_t mask = (1 << mod) - 1;
763  // note - lookup table results in a nearly 10% performance improvement in fill* functions
764  static uint8_t postmask[8] = {0x00, 0x01, 0x03, 0x07, 0x0F, 0x1F, 0x3F, 0x7F };
765  register uint8_t mask = postmask[mod];
766  switch (color)
767  {
768  case WHITE: *pBuf |= mask; break;
769  case BLACK: *pBuf &= ~mask; break;
770  case INVERSE: *pBuf ^= mask; break;
771  }
772  }
