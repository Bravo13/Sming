52 void BMP180::Initialize()
53 {
54  Read2(Reg_CalibrationStart, Reg_CalibrationEnd - Reg_CalibrationStart + 2, buffer);
55  // This data is in Big Endian format from the BMP180.
56  Calibration_AC1 = (buffer[0] << 8) | buffer[1];
57  Calibration_AC2 = (buffer[2] << 8) | buffer[3];
58  Calibration_AC3 = (buffer[4] << 8) | buffer[5];
59  Calibration_AC4 = (buffer[6] << 8) | buffer[7];
60  Calibration_AC5 = (buffer[8] << 8) | buffer[9];
61  Calibration_AC6 = (buffer[10] << 8) | buffer[11];
62  Calibration_B1 = (buffer[12] << 8) | buffer[13];
63  Calibration_B2 = (buffer[14] << 8) | buffer[15];
64  Calibration_MB = (buffer[16] << 8) | buffer[17];
65  Calibration_MC = (buffer[18] << 8) | buffer[19];
66  Calibration_MD = (buffer[20] << 8) | buffer[21];
67 }

95 long BMP180::GetUncompensatedPressure()
96 {
97  long pressure = 0;
98  int loops = Oversample ? 3 : 1;
99 
100  for (int i = 0; i < loops; i++)
101  {
102  // Instruct device to perform a conversion, including the oversampling data.
103  uint8_t CtrlByte = ControlInstruction_MeasurePressure + (OversamplingSetting << 6);
104  Write(Reg_Control, CtrlByte);
105  // Wait for the conversion
106  delay(ConversionWaitTimeMs);
107  // Read the conversion data.
108  uint8_t buffer[3];
109  Read2(Reg_AnalogConverterOutMSB, 3, buffer);
110 
111  // Collect the data (and push back the LSB if we are not sampling them).
112  pressure = ((((long)buffer[0] <<16) | ((long)buffer[1] <<8) | ((long)buffer[2])) >> (8-OversamplingSetting));
113  }
114  return pressure / loops;
115 }

117 float BMP180::CompensateTemperature(int uncompensatedTemperature)
118 {
119  int temperature;
120  int x2;
121  long x1;
122  x1 = (((long)uncompensatedTemperature - (long)Calibration_AC6) * (long)Calibration_AC5) >> 15;
123  x2 = ((long)Calibration_MC << 11) / (x1 + Calibration_MD);
124  int param_b5 = x1 + x2;
125  temperature = (int)((param_b5 + 8) >> 4); /* temperature in 0.1 deg C*/
126  float fTemperature = temperature;
127  fTemperature /= 10.0;
128 
129  // Record this data because it is required by the pressure algorithem.
130  LastTemperatureData = param_b5;
131  LastTemperatureTime = millis();
132 
133  return fTemperature;
134 }

136 long BMP180::CompensatePressure(long uncompensatedPressure)
137 {
138  int msSinceLastTempReading = millis() - LastTemperatureTime;
139  // Check to see if we have old temperature data.
140  if (msSinceLastTempReading > AcceptableTemperatureLatencyForPressure)
141  GetTemperature(); // Refresh the temperature.
142 
143  // Data from the BMP180 datasheet to test algorithm.
144  /*OversamplingSetting = 0;
145  uncompensatedPressure = 23843;
146  LastTemperatureData = 2399;
147  Calibration_AC1 = 408;
148  Calibration_AC2 = -72;
149  Calibration_AC3 = -14383;
150  Calibration_AC4 = 32741;
151  Calibration_AC5 = 32757;
152  Calibration_AC6 = 23153;
153  Calibration_B1 = 6190;
154  Calibration_B2 = 4;
155  Calibration_MB = -32767;
156  Calibration_MC = -8711;
157  Calibration_MD = 2868;*/
158 
159  // Algorithm taken from BMP180 datasheet.
160  long b6 = LastTemperatureData - 4000;
161  long x1 = (Calibration_B2 * (b6 * b6 >> 12)) >> 11;
162  long x2 = Calibration_AC2 * b6 >> 11;
163  long x3 = x1 + x2;
164  long b3 = ((Calibration_AC1 * 4 + x3) << OversamplingSetting) + 2;
165  b3 = b3 >> 2;
166  x1 = Calibration_AC3 * b6 >> 13;
167  x2 = (Calibration_B1 * (b6 * b6 >> 12)) >> 16;
168  x3 = ((x1 + x2) + 2) >> 2;
169  long b4 = Calibration_AC4 * (x3 + 32768) >> 15;
170  unsigned long b7 = (((uncompensatedPressure - b3)) * (50000 >> OversamplingSetting));
171  long p;
172  if (b7 < 0x80000000)
173  {
174  p = ((b7 * 2) / b4);
175  }
176  else
177  {
178  p = ((b7 / b4) * 2);
179  }
180 
181  x1 = (p >> 8) * (p >> 8);
182  x1 = (x1 * 3038) >> 16;
183  x2 = (-7357 * p) >> 16;
184  p = p + ((x1 + x2 + 3791) >> 4);
185 
186  return p;
187 }
