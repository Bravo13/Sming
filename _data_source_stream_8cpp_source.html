162 uint16_t TemplateFileStream::getDataPointer(char** data)
163 {
164  debugf("READ Template (%d)", state);
165 
166  if (state == eTES_StartVar)
167  {
168  if (templateData.contains(varName))
169  {
170  // Return variable value
171  debugf("StartVar %s", varName.c_str());
172  *data = (char*)templateData[varName].c_str();
173  seek(skipBlockSize);
174  varDataPos = 0;
175  state = eTES_SendingVar;
176  return templateData[varName].length();
177  }
178  else
179  {
180  debugf("var %s not found", varName.c_str());
181  state = eTES_Wait;
182  int len = FileStream::getDataPointer(data);
183  return min(len, skipBlockSize);
184  }
185  }
186  else if (state == eTES_SendingVar)
187  {
188  String *val = &templateData[varName];
189  if (varDataPos < val->length())
190  {
191  debugf("continue TRANSFER variable value (not completed)");
192  *data = (char*)val->c_str();
193  *data += varDataPos;
194  return val->length() - varDataPos;
195  }
196  else
197  {
198  debugf("continue to plaint text");
199  state = eTES_Wait;
200  }
201  }
202 
203  int len = FileStream::getDataPointer(data);
204  char* tpl = *data;
205  if (tpl && len > 0)
206  {
207  char* end = tpl + len;
208  char* cur = (char*)memchr(tpl, '{', len);
209  char* lastFound = cur;
210  while (cur != NULL)
211  {
212  lastFound = cur;
213  char* p = cur + 1;
214  for (; p < end; p++)
215  {
216  if (isspace(*p))
217  break; // Not a var name
218  else if (p - cur > TEMPLATE_MAX_VAR_NAME_LEN)
219  break; // To long for var name
220  else if (*p == '{')
221  break; // New start..
222 
223  if (*p == '}')
224  {
225  int block = p - cur + 1;
226  char varname[TEMPLATE_MAX_VAR_NAME_LEN + 1] = {0};
227  memcpy(varname, cur + 1, p - cur - 1); // name without { and }
228  varName = varname;
229  state = eTES_Found;
230  varWaitSize = cur - tpl;
231  debugf("found var: %s, at %d (%d) - %d, send size %d", varName.c_str(), cur - tpl + 1, cur - tpl + getPos(), p - tpl, cur - tpl);
232  skipBlockSize = block;
233  return cur - tpl; // return only plain text from template without our variable
234  }
235  }
236  cur = (char*)memchr(p, '{', len - (p - tpl)); // continue searching..
237  }
238  if (lastFound != NULL && (lastFound - tpl) > (len - TEMPLATE_MAX_VAR_NAME_LEN))
239  {
240  debugf("trim end to %d from %d", lastFound - tpl, len);
241  len = lastFound - tpl; // It can be a incomplete variable name. Don't split it!
242  }
243  }
244 
245  debugf("plain template text pos: %d, len: %d", getPos(), len);
246  return len;
247 }
