11 void pinMode(uint16_t pin, uint8_t mode)
12 {
13  if (pin < 16)
14  {
15  // Set as GPIO
16  PIN_FUNC_SELECT((EspDigitalPins[pin].mux), (EspDigitalPins[pin].gpioFunc));
17 
18  // Default Pull-up
19  pullup(pin);
20 
21  // Switch to Input or Output
22  if (mode == INPUT)
23  GPIO_REG_WRITE(GPIO_ENABLE_W1TC_ADDRESS, (1<<pin));
24  else
25  GPIO_REG_WRITE(GPIO_ENABLE_W1TS_ADDRESS, (1<<pin));
26  }
27  else if (pin == 16)
28  {
29  if (mode == INPUT)
30  {
31  WRITE_PERI_REG(PAD_XPD_DCDC_CONF,
32  (READ_PERI_REG(PAD_XPD_DCDC_CONF) & 0xffffffbc) | (uint32)0x1); // mux configuration for XPD_DCDC and rtc_gpio0 connection
33 
34  WRITE_PERI_REG(RTC_GPIO_CONF,
35  (READ_PERI_REG(RTC_GPIO_CONF) & (uint32)0xfffffffe) | (uint32)0x0); //mux configuration for out enable
36 
37  WRITE_PERI_REG(RTC_GPIO_ENABLE,
38  READ_PERI_REG(RTC_GPIO_ENABLE) & (uint32)0xfffffffe); //out disable
39  }
40  else
41  {
42  WRITE_PERI_REG(PAD_XPD_DCDC_CONF,
43  (READ_PERI_REG(PAD_XPD_DCDC_CONF) & 0xffffffbc) | (uint32)0x1); // mux configuration for XPD_DCDC to output rtc_gpio0
44 
45  WRITE_PERI_REG(RTC_GPIO_CONF,
46  (READ_PERI_REG(RTC_GPIO_CONF) & (uint32)0xfffffffe) | (uint32)0x0); //mux configuration for out enable
47 
48  WRITE_PERI_REG(RTC_GPIO_ENABLE,
49  (READ_PERI_REG(RTC_GPIO_ENABLE) & (uint32)0xfffffffe) | (uint32)0x1); //out enable
50  }
51  } else
52  SYSTEM_ERROR("No pin %d, can't set mode", pin); // NO PIN!
53 }

96 /* Measures the length (in microseconds) of a pulse on the pin; state is HIGH
97  * or LOW, the type of pulse to measure. Works on pulses from 2-3 microseconds
98  * to 3 minutes in length, but must be called at least a few dozen microseconds
99  * before the start of the pulse. */
100 unsigned long pulseIn(uint16_t pin, uint8_t state, unsigned long timeout)
101 {
102  // cache the port and bit of the pin in order to speed up the
103  // pulse width measuring loop and achieve finer resolution. calling
104  // digitalRead() instead yields much coarser resolution.
105  uint8_t bit = digitalPinToBitMask(pin);
106  uint8_t port = digitalPinToPort(pin);
107  uint8_t stateMask = (state ? bit : 0);
108  unsigned long width = 0; // keep initialization out of time critical area
109 
110  // convert the timeout from microseconds to a number of times through
111  // the initial loop; it takes 16 clock cycles per iteration.
112  unsigned long numloops = 0;
113  unsigned long maxloops = microsecondsToClockCycles(timeout) / 16;
114 
115  // wait for any previous pulse to end
116  while ((*portInputRegister(port) & bit) == stateMask)
117  if (numloops++ == maxloops)
118  return 0;
119 
120  // wait for the pulse to start
121  while ((*portInputRegister(port) & bit) != stateMask)
122  if (numloops++ == maxloops)
123  return 0;
124 
125  // wait for the pulse to stop
126  while ((*portInputRegister(port) & bit) == stateMask) {
127  if (numloops++ == maxloops)
128  return 0;
129  width++;
130  }
131 
132  // convert the reading to microseconds. The loop has been determined
133  // to be 20 clock cycles long and have about 16 clocks between the edge
134  // and the start of the loop. There will be some error introduced by
135  // the interrupt handlers.
136 
137  return clockCyclesToMicroseconds(width * 21 + 16);
138 }
