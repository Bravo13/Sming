159 {
160  int line, nextLine;

173  String value = NetUtils::pbufStrCopy(buf, delim + 1,

178  }

180  }
181  line = nextLine + 2;
182  } while (nextLine != -1 && nextLine < headerEnd);

185 void HttpClient::writeRawData(pbuf* buf, int startPos)
186 {
187  switch (mode)
188  {
189  case eHCM_String:
190  {
191  responseStringData += NetUtils::pbufStrCopy(buf, startPos,
192  buf->tot_len - startPos);
193  break;
194  }
195  case eHCM_File:
196  {
197  pbuf *cur = buf;
198  while (cur != NULL && cur->len > 0 && !writeError)
199  {
200  char* ptr = (char*) cur->payload + startPos;
201  int len = cur->len - startPos;
202  int res = fileWrite(saveFile, ptr, len);
203  writeError |= (res < 0);
204  cur = cur->next;
205  startPos = 0;
206  }
207 
208  if (writeError)
209  close();
210  }
211  }

214 err_t HttpClient::onReceive(pbuf *buf)
215 {
216  if (buf == NULL)
217  {
218  // Disconnected, close it
219  TcpClient::onReceive(buf);
220  }
221  else
222  {
223  int startPos = 0;
224  if (waitParse)
225  {
226  String http = NetUtils::pbufStrCopy(buf, 0, 4);
227  http.toUpperCase();
228  if (http == "HTTP" && code == 0)
229  {
230  int codeStart = NetUtils::pbufFindChar(buf, ' ', 4);
231  int codeEnd = NetUtils::pbufFindChar(buf, ' ', codeStart + 1);
232  if (codeStart != -1 && codeEnd != -1 && codeEnd - codeStart < 5)
233  {
234  String strCode = NetUtils::pbufStrCopy(buf, codeStart, codeEnd);
235  code = strCode.toInt();
236  }
237  else
238  code = 0;
239  }
240  int headerEnd = NetUtils::pbufFindStr(buf, "\r\n\r\n");
241  if (headerEnd != -1)
242  {
243  debugf("Header pos: %d", headerEnd);
244  startPos = headerEnd + 4;
245  waitParse = false;
246  if (headerEnd < NETWORK_MAX_HTTP_PARSING_LEN)
247  parseHeaders(buf, headerEnd);
248  }
249  }
250 
251  writeRawData(buf, startPos);
252 
253  // Fire ReadyToSend callback
254  TcpClient::onReceive(buf);
255  }
256 
257  return ERR_OK;
258 }
