120 int I2CIO::write ( uint8_t value )
121 {
122  int status = 0;
123 
124  if ( _initialised )
125  {
126  // Only write HIGH the values of the ports that have been initialised as
127  // outputs updating the output shadow of the device
128  _shadow = ( value & ~(_dirMask) );
129 
130  Wire.beginTransmission ( _i2cAddr );
131  Wire.write ( _shadow );
132  status = Wire.endTransmission ();
133  }
134  return ( (status == 0) );
135 }

139 uint8_t I2CIO::digitalRead ( uint8_t pin )
140 {
141  uint8_t pinVal = 0;
142 
143  // Check if initialised and that the pin is within range of the device
144  // -------------------------------------------------------------------
145  if ( ( _initialised ) && ( pin <= 7 ) )
146  {
147  // Remove the values which are not inputs and get the value of the pin
148  pinVal = this->read() & _dirMask;
149  pinVal = ( pinVal >> pin ) & 0x01; // Get the pin value
150  }
151  return (pinVal);

157 {
158  uint8_t writeVal;
159  int status = 0;
160 
161  // Check if initialised and that the pin is within range of the device
162  // -------------------------------------------------------------------
163  if ( ( _initialised ) && ( pin <= 7 ) )
164  {
165  // Only write to HIGH the port if the port has been configured as
166  // an OUTPUT pin. Add the new state of the pin to the shadow
167  writeVal = ( 1 << pin ) & ~_dirMask;
168  if ( level == HIGH )
169  {
170  _shadow |= writeVal;
171 
172  }
173  else
174  {
175  _shadow &= ~writeVal;
176  }
177  status = this->write ( _shadow );
178  }
179  return ( status );
180 }
