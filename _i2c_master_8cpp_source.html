53 uint8_t SoftI2cMaster::read(uint8_t last) {
54  uint8_t b = 0;
55  // make sure pull-up enabled
56  digitalWrite(sdaPin_, HIGH);
57  pinMode(sdaPin_, INPUT);
58  // read byte
59  for (uint8_t i = 0; i < 8; i++) {
60  // don't change this loop unless you verify the change with a scope
61  b <<= 1;
62  delayMicroseconds(I2C_DELAY_USEC);
63  digitalWrite(sclPin_, HIGH);
64  if (digitalRead(sdaPin_)) b |= 1;
65  digitalWrite(sclPin_, LOW);
66  }
67  // send Ack or Nak
68  pinMode(sdaPin_, OUTPUT);
69  digitalWrite(sdaPin_, last);
70  digitalWrite(sclPin_, HIGH);
71  delayMicroseconds(I2C_DELAY_USEC);
72  digitalWrite(sclPin_, LOW);
73  digitalWrite(sdaPin_, LOW);
74  return b;
75 }

120 bool SoftI2cMaster::write(uint8_t data) {
121  // write byte
122  for (uint8_t m = 0X80; m != 0; m >>= 1) {
123  // don't change this loop unless you verify the change with a scope
124  digitalWrite(sdaPin_, m & data);
125  digitalWrite(sclPin_, HIGH);
126  delayMicroseconds(I2C_DELAY_USEC);
127  digitalWrite(sclPin_, LOW);
128  }
129  // get Ack or Nak
130  pinMode(sdaPin_, INPUT);
131  // enable pullup
132  digitalWrite(sdaPin_, HIGH);
133  digitalWrite(sclPin_, HIGH);
134  uint8_t rtn = digitalRead(sdaPin_);
135  digitalWrite(sclPin_, LOW);
136  pinMode(sdaPin_, OUTPUT);
137  digitalWrite(sdaPin_, LOW);
138  return rtn == 0;
139 }
