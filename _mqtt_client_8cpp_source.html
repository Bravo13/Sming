122 err_t MqttClient::onReceive(pbuf *buf)
123 {
124  if (buf == NULL)
125  {
126  // Disconnected, close it
127  TcpClient::onReceive(buf);
128  }
129  else
130  {
131  if (buf->len < 1)
132  {
133  // Bad packet?
134  debugf("> MQTT WRONG PACKET? (len: %d)", buf->len);
135  close();
136  return ERR_OK;
137  }
138 
139  int received = 0;
140  while (received < buf->tot_len)
141  {
142  int type = 0;
143  if (waitingSize == 0)
144  {
145  // It's begining of new packet
146  int pos = received;
147  if (posHeader == 0)
148  {
149  //debugf("start posHeader");
150  pbuf_copy_partial(buf, &buffer[posHeader], 1, pos);
151  pos++;
152  posHeader = 1;
153  }
154  while (posHeader > 0 && pos < buf->tot_len)
155  {
156  //debugf("add posHeader");
157  pbuf_copy_partial(buf, &buffer[posHeader], 1, pos);
158  if ((buffer[posHeader] & 128) == 0)
159  posHeader = 0; // Remaining Length ended
160  else
161  posHeader++;
162  pos++;
163  }
164 
165  if (posHeader == 0)
166  {
167  //debugf("start len calc");
168  // Remaining Length field processed
169  uint16_t rem_len = mqtt_parse_rem_len(buffer);
170  uint8_t rem_len_bytes = mqtt_num_rem_len_bytes(buffer);
171 
172  // total packet length = remaining length + byte 1 of fixed header + remaning length part of fixed header
173  waitingSize = rem_len + rem_len_bytes + 1;
174 
175  type = MQTTParseMessageType(buffer);
176  debugPrintResponseType(type, waitingSize);
177 
178  // Prevent overflow
179  if (waitingSize < MQTT_MAX_BUFFER_SIZE)
180  {
181  current = buffer;
182  buffer[waitingSize] = 0;
183  }
184  else
185  current = NULL;
186  }
187  else
188  continue;
189  }
190 
191  int available = min(waitingSize, buf->tot_len - received);
192  waitingSize -= available;
193  if (current != NULL)
194  {
195  pbuf_copy_partial(buf, current, available, received);
196  current += available;
197 
198  if (waitingSize == 0)
199  {
200  // Full packet received
201  if(type == MQTT_MSG_PUBLISH)
202  {
203  const uint8_t *ptrTopic, *ptrMsg;
204  uint16_t lenTopic, lenMsg;
205  lenTopic = mqtt_parse_pub_topic_ptr(buffer, &ptrTopic);
206  lenMsg = mqtt_parse_pub_msg_ptr(buffer, &ptrMsg);
207  // Additional check for wrong packet/parsing error
208  if (lenTopic + lenMsg < MQTT_MAX_BUFFER_SIZE)
209  {
210  debugf("%d: %d\n", lenTopic, lenMsg);
211  String topic, msg;
212  topic.setString((char*)ptrTopic, lenTopic);
213  msg.setString((char*)ptrMsg, lenMsg);
214  if (callback)
215  callback(topic, msg);
216  }
217  else
218  {
219  debugf("WRONG SIZES: %d: %d", lenTopic, lenMsg);
220  }
221  }
222  }
223  }
224  else
225  debugf("SKIP: %d (%d)", available, waitingSize + available); // To large!
226  received += available;
227  }
228 
229  // Fire ReadyToSend callback
230  TcpClient::onReceive(buf);
231  }
232 
233  return ERR_OK;
234 }
