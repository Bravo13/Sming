89 void NtpClient::requestTime()
90 {
91  if (!WifiStation.isConnected())
92  {
93  connectionTimer.initializeMs(1000, Delegate<void()>(&NtpClient::requestTime, this)).startOnce();
94  return;
95  }
96  if (serverAddress.isNull())
97  {
98  if (!resolveServer())
99  {
100  return;
101  }
102  }
103 
104 // connect to current active serverAddress, on NTP_PORT
105  this->connect(serverAddress,NTP_PORT);
106 
107  uint8_t packet[NTP_PACKET_SIZE];
108 
109  // Setup the NTP request packet
110  memset(packet, 0, NTP_PACKET_SIZE);
111 
112  // These are the only required values for a SNTP request. See page 14:
113  // https://tools.ietf.org/html/rfc4330
114  // However size of packet should still be 48 bytes.
115  packet[0] = (NTP_VERSION << 3 | 0x03); // LI (0 = no warning), Protocol version (4), Client mode (3)
116  packet[1] = 0; // Stratum, or type of clock, unspecified.
117 
118 // Send to server, serverAddress & port is set in connect
119  NtpClient::send((char*) packet, NTP_PACKET_SIZE);
120 }

157 void NtpClient::onReceive(pbuf *buf, IPAddress remoteIP, uint16_t remotePort)
158 {
159  // We do some basic check to see if it really is a ntp packet we receive.
160  // NTP version should be set to same as we used to send, NTP_VERSION
161  // NTP_VERSION 3 has time in same location so accept that too
162  // Mode should be set to NTP_MODE_SERVER
163 
164  uint8_t versionMode = pbuf_get_at(buf, 0);
165  uint8_t ver = (versionMode & 0b00111000) >> 3;
166  uint8_t mode = (versionMode & 0x07);
167 
168  if (mode == NTP_MODE_SERVER && (ver == NTP_VERSION || ver == (NTP_VERSION -1)))
169  {
170  //Most likely a correct NTP packet received.
171 
172  uint8_t data[4];
173  pbuf_copy_partial(buf, data, 4, 40); // Copy only timestamp.
174 
175  uint32_t timestamp = (data[0] << 24 | data[1] << 16 | data[2] << 8
176  | data[3]);
177 
178  // Unix time starts on Jan 1 1970, subtract 70 years:
179  uint32_t epoch = timestamp - 0x83AA7E80;
180 
181  if (autoUpdateSystemClock)
182  {
183  SystemClock.setTime(epoch, eSCUtc); // update systemclock utc value
184  }
185 
186  if (onCompleted != NULL)
187  {
188  this->onCompleted(*this, epoch);
189  }
190  if (delegateCompleted)
191  {
192  this->delegateCompleted(*this, epoch);
193  }
194  }
195 }
