20 void SPIClass::begin()
21 {
22  //bit9 of PERIPHS_IO_MUX should be cleared when HSPI clock doesn't equal CPU clock
23  //bit8 of PERIPHS_IO_MUX should be cleared when SPI clock doesn't equal CPU clock
24  if (id == SPI_ID_MAIN)
25  {
26  WRITE_PERI_REG(PERIPHS_IO_MUX, 0x005); //clear bit9,and bit8
27 
28  PIN_FUNC_SELECT(PERIPHS_IO_MUX_SD_CLK_U, 1); //configure io to spi mode
29  PIN_FUNC_SELECT(PERIPHS_IO_MUX_SD_CMD_U, 1); //configure io to spi mode
30  PIN_FUNC_SELECT(PERIPHS_IO_MUX_SD_DATA0_U, 1); //configure io to spi mode
31  PIN_FUNC_SELECT(PERIPHS_IO_MUX_SD_DATA1_U, 1); //configure io to spi mode
32  }
33  else if (id == SPI_ID_HSPI)
34  {
35  WRITE_PERI_REG(PERIPHS_IO_MUX, 0x105); //clear bit9
36 
37  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTDI_U, 2); // HSPIQ MISO == GPIO12
38  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTCK_U, 2); // HSPID MOSI == GPIO13
39  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTMS_U, 2); // CLK == GPIO14
40  }
41  else
42  SYSTEM_ERROR("UNSUPPORTED SPI id: %d", id);
43 
44 
45  // SPI clock = CPU clock / 10 / 4
46  // time length HIGHT level = (CPU clock / 10 / 2) ^ -1,
47  // time length LOW level = (CPU clock / 10 / 2) ^ -1
48  WRITE_PERI_REG(SPI_FLASH_CLOCK(id),
49  ((1 & SPI_CLKDIV_PRE) << SPI_CLKDIV_PRE_S) |
50  ((3 & SPI_CLKCNT_N) << SPI_CLKCNT_N_S) |
51  ((1 & SPI_CLKCNT_H) << SPI_CLKCNT_H_S) |
52  ((3 & SPI_CLKCNT_L) << SPI_CLKCNT_L_S));
53 }

55 void SPIClass::writeData(uint8_t * data, uint8_t numberByte)
56 {
57  uint8_t i = 0;
58  uint8_t shift = 0;
59  uint32_t *buffer = (uint32_t *)SPI_FLASH_C0(id);
60  for (i = 0; i < numberByte; ++i)
61  {
62  if (shift >= 32)
63  {
64  shift = 0;
65  buffer += 1;
66  }
67  *buffer &= ~( 0xFFUL << shift );
68  *buffer |= ((uint32_t)data[i]) << shift;
69  shift += 8;
70  }
71 }

73 void SPIClass::readData(uint8_t * data, uint8_t numberByte)
74 {
75  uint8_t i = 0;
76  uint8_t shift = 0;
77  uint32_t buffer = 0;
78 
79  for (i = 0; i < numberByte; ++i)
80  {
81  if ( (i % 4 == 0) )
82  {
83  buffer = ( *( (uint32_t *)SPI_FLASH_C0(id) + i/4 ) );
84  shift = 0;
85  }
86  data[i] = (buffer >> shift) & 0xFF;
87  shift += 8;
88  }
89 }

95 void SPIClass::transfer(uint8_t * data, uint8_t count)
96 {
97  uint32_t regvalue = 0;
98  uint16_t numberBit = 0;
99 
100  while (READ_PERI_REG(SPI_FLASH_CMD(id))&SPI_FLASH_USR);
101 
102  regvalue |= SPI_FLASH_DOUT | SPI_DOUTDIN | SPI_CK_I_EDGE;
103  regvalue &= ~(BIT2 | SPI_FLASH_USR_ADDR | SPI_FLASH_USR_DUMMY | SPI_FLASH_USR_DIN | SPI_USR_COMMAND); //clear bit 2 see example IoT_Demo
104  WRITE_PERI_REG(SPI_FLASH_USER(id), regvalue);
105 
106  numberBit = count * 8 - 1;
107 
108  WRITE_PERI_REG(SPI_FLASH_USER1(id),
109  ( (numberBit & SPI_USR_OUT_BITLEN) << SPI_USR_OUT_BITLEN_S ) |
110  ( (numberBit & SPI_USR_DIN_BITLEN) << SPI_USR_DIN_BITLEN_S ) );
111 
112  writeData(data, count);
113 
114  SET_PERI_REG_MASK(SPI_FLASH_CMD(id), SPI_FLASH_USR); // send
115 
116  while (READ_PERI_REG(SPI_FLASH_CMD(id)) & SPI_FLASH_USR);
117 
118  readData(data, count);
119 }
