102 uint16_t Ultrasonic::temp(uint16_t baseDist, uint16_t baseTemp, uint16_t samples)
103 {

105  for (int i = 0; i < samples; i++)
106  {
107  dist += rangeCM();
108  delay(60);

110 
111  // avg dist

114  // c = sqrt(X*R*T)

116  float speedOfSound = baseDist / dist * root(XR * (K + baseTemp));
