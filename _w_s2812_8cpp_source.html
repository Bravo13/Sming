7 // The ICACHE_FLASH_ATTR is there to trick the compiler and get the very first pulse width correct.
8 static void ICACHE_FLASH_ATTR send_ws_0(uint8_t gpio)
9 {
10  uint8_t i;
11  i = 4; while (i--) GPIO_REG_WRITE(GPIO_OUT_W1TS_ADDRESS, 1 << gpio);
12  i = 9; while (i--) GPIO_REG_WRITE(GPIO_OUT_W1TC_ADDRESS, 1 << gpio);
13 }

22 // Byte triples in the buffer are interpreted as R G B values and sent to the hardware as G R B.
23 int ICACHE_FLASH_ATTR ws2812_writergb(uint8_t gpio, char *buffer, size_t length)
24 {
25  // Initialize the output pin:
26  pinMode(gpio, OUTPUT);
27  digitalWrite(gpio, 0);
28 
29  // Ignore incomplete Byte triples at the end of buffer:
30  length -= length % 3;
31 
32  // Rearrange R G B values to G R B order needed by WS2812 LEDs:
33  size_t i;
34  for (i = 0; i < length; i += 3) {
35  const char r = buffer[i];
36  const char g = buffer[i + 1];
37  buffer[i] = g;
38  buffer[i + 1] = r;
39  }
40 
41  // Do not remove these:
42  os_delay_us(1);
43  os_delay_us(1);
44 
45  // Send the buffer:
46  os_intr_lock();
47  const char * const end = buffer + length;
48  while (buffer != end) {
49  uint8_t mask = 0x80;
50  while (mask) {
51  (*buffer & mask) ? send_ws_1(gpio) : send_ws_0(gpio);
52  mask >>= 1;
53  }
54  ++buffer;
55  }
56  os_intr_unlock();
57 }
