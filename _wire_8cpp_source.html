72 uint8_t TwoWire::pushData()
73 {
74  if (txLen == -1) return 1; // data too long to fit in transmit buffer
75 
76  if (!master->start(targetAddress | I2C_WRITE)) return 2; // received NACK on transmit of address
77 
78  for (int i = 0; i < txLen; i++)
79  if (!master->write(txBuf[i]))
80  return 3; // received NACK on transmit of data
81 
82  targetAddress = -1;
83  txLen = 0;
84  return 0;
85 }

87 uint8_t TwoWire::requestFrom(int address, int quantity, bool sendStop /* = true*/)
88 {
89  rxPos = 0;
90  rxLen = 0;
91 
92  if (!master->start(((uint8_t)(address << 1)) | I2C_READ)) return 0; // received NACK on transmit of address
93 
94  for (int i = 0; i < quantity; i++)
95  rxBuf[rxLen++] = master->read(sendStop && (quantity == i + 1));
96 
97  return quantity;
98 }
