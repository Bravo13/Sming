111 // Garbage collecting examines all pages in a block which and sums up
112 // to a block score. Deleted pages normally gives positive score and
113 // used pages normally gives a negative score (as these must be moved).
114 // To have a fair wear-leveling, the erase age is also included in score,
115 // whose factor normally is the most positive.
116 // The larger the score, the more likely it is that the block will
117 // picked for garbage collection.
118 
119 // Garbage collecting heuristics - weight used for deleted pages.
120 #ifndef SPIFFS_GC_HEUR_W_DELET
121 #define SPIFFS_GC_HEUR_W_DELET (5)
122 #endif
123 // Garbage collecting heuristics - weight used for used pages.
124 #ifndef SPIFFS_GC_HEUR_W_USED
125 #define SPIFFS_GC_HEUR_W_USED (-1)
126 #endif
127 // Garbage collecting heuristics - weight used for time between
128 // last erased and erase of this block.
