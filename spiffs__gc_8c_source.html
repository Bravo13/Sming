4 // Erases a logical block and updates the erase counter.
5 // If cache is enabled, all pages that might be cached in this block
6 // is dropped.
7 static s32_t spiffs_gc_erase_block(
8  spiffs *fs,
9  spiffs_block_ix bix) {
10  s32_t res;
11  u32_t addr = SPIFFS_BLOCK_TO_PADDR(fs, bix);
12  s32_t size = SPIFFS_CFG_LOG_BLOCK_SZ(fs);
13 
14  SPIFFS_GC_DBG("gc: erase block %i\n", bix);
15 
16  // here we ignore res, just try erasing the block
17  while (size > 0) {
18  SPIFFS_GC_DBG("gc: erase %08x:%08x\n", addr, SPIFFS_CFG_PHYS_ERASE_SZ(fs));
19  (void)fs->cfg.hal_erase_f(addr, SPIFFS_CFG_PHYS_ERASE_SZ(fs));
20  addr += SPIFFS_CFG_PHYS_ERASE_SZ(fs);
21  size -= SPIFFS_CFG_PHYS_ERASE_SZ(fs);
22  }
23  fs->free_blocks++;
24 
25  // register erase count for this block
26  res = _spiffs_wr(fs, SPIFFS_OP_C_WRTHRU | SPIFFS_OP_T_OBJ_LU2, 0,
27  SPIFFS_ERASE_COUNT_PADDR(fs, bix),
28  sizeof(spiffs_obj_id), (u8_t *)&fs->max_erase_count);
29  SPIFFS_CHECK_RES(res);
30 
31  fs->max_erase_count++;
32  if (fs->max_erase_count == SPIFFS_OBJ_ID_IX_FLAG) {
33  fs->max_erase_count = 0;
34  }
35 
36 #if SPIFFS_CACHE
37  {
38  u32_t i;
39  for (i = 0; i < SPIFFS_PAGES_PER_BLOCK(fs); i++) {
40  spiffs_cache_drop_page(fs, SPIFFS_PAGE_FOR_BLOCK(fs, bix) + i);
41  }
42  }
43 #endif
44  return res;
45 }

47 // Searches for blocks where all entries are deleted - if one is found,
48 // the block is erased. Compared to the non-quick gc, the quick one ensures
49 // that no updates are needed on existing objects on pages that are erased.
50 s32_t spiffs_gc_quick(
51  spiffs *fs) {
52  s32_t res = SPIFFS_OK;
53  u32_t blocks = fs->block_count;
54  spiffs_block_ix cur_block = 0;
55  u32_t cur_block_addr = 0;
56  int cur_entry = 0;
57  spiffs_obj_id *obj_lu_buf = (spiffs_obj_id *)fs->lu_work;
58 
59  SPIFFS_GC_DBG("gc_quick: running\n", cur_block);
60 #if SPIFFS_GC_STATS
61  fs->stats_gc_runs++;
62 #endif
63 
64  int entries_per_page = (SPIFFS_CFG_LOG_PAGE_SZ(fs) / sizeof(spiffs_obj_id));
65 
66  // find fully deleted blocks
67  // check each block
68  while (res == SPIFFS_OK && blocks--) {
69  u16_t deleted_pages_in_block = 0;
70 
71  int obj_lookup_page = 0;
72  // check each object lookup page
73  while (res == SPIFFS_OK && obj_lookup_page < (int)SPIFFS_OBJ_LOOKUP_PAGES(fs)) {
74  int entry_offset = obj_lookup_page * entries_per_page;
75  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU | SPIFFS_OP_C_READ,
76  0, cur_block_addr + SPIFFS_PAGE_TO_PADDR(fs, obj_lookup_page), SPIFFS_CFG_LOG_PAGE_SZ(fs), fs->lu_work);
77  // check each entry
78  while (res == SPIFFS_OK &&
79  cur_entry - entry_offset < entries_per_page &&
80  cur_entry < (int)(SPIFFS_PAGES_PER_BLOCK(fs)-SPIFFS_OBJ_LOOKUP_PAGES(fs))) {
81  spiffs_obj_id obj_id = obj_lu_buf[cur_entry-entry_offset];
82  if (obj_id == SPIFFS_OBJ_ID_DELETED) {
83  deleted_pages_in_block++;
84  } else if (obj_id == SPIFFS_OBJ_ID_FREE) {
85  // kill scan, go for next block
86  obj_lookup_page = SPIFFS_OBJ_LOOKUP_PAGES(fs);
87  res = 1; // kill object lu loop
88  break;
89  } else {
90  // kill scan, go for next block
91  obj_lookup_page = SPIFFS_OBJ_LOOKUP_PAGES(fs);
92  res = 1; // kill object lu loop
93  break;
94  }
95  cur_entry++;
96  } // per entry
97  obj_lookup_page++;
98  } // per object lookup page
99  if (res == 1) res = SPIFFS_OK;
100 
101  if (res == SPIFFS_OK && deleted_pages_in_block == SPIFFS_PAGES_PER_BLOCK(fs)-SPIFFS_OBJ_LOOKUP_PAGES(fs)) {
102  // found a fully deleted block
103  fs->stats_p_deleted -= deleted_pages_in_block;
104  res = spiffs_gc_erase_block(fs, cur_block);
105  return res;
106  }
107 
108  cur_entry = 0;
109  cur_block++;
110  cur_block_addr += SPIFFS_CFG_LOG_BLOCK_SZ(fs);
111  } // per block
112 
113  return res;
114 }

118 s32_t spiffs_gc_check(
119  spiffs *fs,
120  u32_t len) {
121  s32_t res;
122  u32_t free_pages =
123  (SPIFFS_PAGES_PER_BLOCK(fs) - SPIFFS_OBJ_LOOKUP_PAGES(fs)) * fs->block_count
124  - fs->stats_p_allocated - fs->stats_p_deleted;
125  int tries = 0;
126 
127  if (fs->free_blocks > 3 &&
128  len < free_pages * SPIFFS_DATA_PAGE_SIZE(fs)) {
129  return SPIFFS_OK;
130  }
131 
132  //printf("gcing started %i dirty, blocks %i free, want %i bytes\n", fs->stats_p_allocated + fs->stats_p_deleted, fs->free_blocks, len);
133 
134  do {
135  SPIFFS_GC_DBG("\ngc_check #%i: run gc free_blocks:%i pfree:%i pallo:%i pdele:%i [%i] len:%i of %i\n",
136  tries,
137  fs->free_blocks, free_pages, fs->stats_p_allocated, fs->stats_p_deleted, (free_pages+fs->stats_p_allocated+fs->stats_p_deleted),
138  len, free_pages*SPIFFS_DATA_PAGE_SIZE(fs));
139 
140  spiffs_block_ix *cands;
141  int count;
142  spiffs_block_ix cand;
143  res = spiffs_gc_find_candidate(fs, &cands, &count);
144  SPIFFS_CHECK_RES(res);
145  if (count == 0) {
146  SPIFFS_GC_DBG("gc_check: no candidates, return\n");
147  return res;
148  }
149 #if SPIFFS_GC_STATS
150  fs->stats_gc_runs++;
151 #endif
152  cand = cands[0];
153  fs->cleaning = 1;
154  //printf("gcing: cleaning block %i\n", cand);
155  res = spiffs_gc_clean(fs, cand);
156  fs->cleaning = 0;
157  if (res < 0) {
158  SPIFFS_GC_DBG("gc_check: cleaning block %i, result %i\n", cand, res);
159  } else {
160  SPIFFS_GC_DBG("gc_check: cleaning block %i, result %i\n", cand, res);
161  }
162  SPIFFS_CHECK_RES(res);
163 
164  res = spiffs_gc_erase_page_stats(fs, cand);
165  SPIFFS_CHECK_RES(res);
166 
167  res = spiffs_gc_erase_block(fs, cand);
168  SPIFFS_CHECK_RES(res);
169

171  (SPIFFS_PAGES_PER_BLOCK(fs) - SPIFFS_OBJ_LOOKUP_PAGES(fs)) * fs->block_count
172  - fs->stats_p_allocated - fs->stats_p_deleted;
173 
174  } while (++tries < SPIFFS_GC_MAX_RUNS && (fs->free_blocks <= 2 ||
175  len > free_pages*SPIFFS_DATA_PAGE_SIZE(fs)));
176  SPIFFS_GC_DBG("gc_check: finished\n");
177 
178  //printf("gcing finished %i dirty, blocks %i free, %i pages free, %i tries, res %i\n",
179  // fs->stats_p_allocated + fs->stats_p_deleted,
180  // fs->free_blocks, free_pages, tries, res);
181 
182  return res;
183 }

222 // Finds block candidates to erase
223 s32_t spiffs_gc_find_candidate(
224  spiffs *fs,
225  spiffs_block_ix **block_candidates,
226  int *candidate_count) {
227  s32_t res = SPIFFS_OK;
228  u32_t blocks = fs->block_count;
229  spiffs_block_ix cur_block = 0;
230  u32_t cur_block_addr = 0;
231  spiffs_obj_id *obj_lu_buf = (spiffs_obj_id *)fs->lu_work;
232  int cur_entry = 0;
233 
234  // using fs->work area as sorted candidate memory, (spiffs_block_ix)cand_bix/(s32_t)score
235  int max_candidates = MIN(fs->block_count, (SPIFFS_CFG_LOG_PAGE_SZ(fs)-8)/(sizeof(spiffs_block_ix) + sizeof(s32_t)));
236  *candidate_count = 0;
237  c_memset(fs->work, 0xff, SPIFFS_CFG_LOG_PAGE_SZ(fs));
238 
239  // divide up work area into block indices and scores
240  // todo alignment?
241  spiffs_block_ix *cand_blocks = (spiffs_block_ix *)fs->work;
242  s32_t *cand_scores = (s32_t *)(fs->work + max_candidates * sizeof(spiffs_block_ix));
243 
244  *block_candidates = cand_blocks;
245 
246  int entries_per_page = (SPIFFS_CFG_LOG_PAGE_SZ(fs) / sizeof(spiffs_obj_id));
247 
248  // check each block
249  while (res == SPIFFS_OK && blocks--) {
250  u16_t deleted_pages_in_block = 0;
251  u16_t used_pages_in_block = 0;
252 
253  int obj_lookup_page = 0;
254  // check each object lookup page
255  while (res == SPIFFS_OK && obj_lookup_page < (int)SPIFFS_OBJ_LOOKUP_PAGES(fs)) {
256  int entry_offset = obj_lookup_page * entries_per_page;
257  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU | SPIFFS_OP_C_READ,
258  0, cur_block_addr + SPIFFS_PAGE_TO_PADDR(fs, obj_lookup_page), SPIFFS_CFG_LOG_PAGE_SZ(fs), fs->lu_work);
259  // check each entry
260  while (res == SPIFFS_OK &&
261  cur_entry - entry_offset < entries_per_page &&
262  cur_entry < (int)(SPIFFS_PAGES_PER_BLOCK(fs)-SPIFFS_OBJ_LOOKUP_PAGES(fs))) {
263  spiffs_obj_id obj_id = obj_lu_buf[cur_entry-entry_offset];
264  if (obj_id == SPIFFS_OBJ_ID_FREE) {
265  // when a free entry is encountered, scan logic ensures that all following entries are free also
266  res = 1; // kill object lu loop
267  break;
268  } else if (obj_id == SPIFFS_OBJ_ID_DELETED) {
269  deleted_pages_in_block++;
270  } else {
271  used_pages_in_block++;
272  }
273  cur_entry++;
274  } // per entry
275  obj_lookup_page++;
276  } // per object lookup page
277  if (res == 1) res = SPIFFS_OK;
278 
279  // calculate score and insert into candidate table
280  // stoneage sort, but probably not so many blocks
281  if (res == SPIFFS_OK && deleted_pages_in_block > 0) {
282  // read erase count
283  spiffs_obj_id erase_count;
284  res = _spiffs_rd(fs, SPIFFS_OP_C_READ | SPIFFS_OP_T_OBJ_LU2, 0,
285  SPIFFS_ERASE_COUNT_PADDR(fs, cur_block),
286  sizeof(spiffs_obj_id), (u8_t *)&erase_count);
287  SPIFFS_CHECK_RES(res);
288 
289  spiffs_obj_id erase_age;
290  if (fs->max_erase_count > erase_count) {
291  erase_age = fs->max_erase_count - erase_count;
292  } else {
293  erase_age = SPIFFS_OBJ_ID_FREE - (erase_count - fs->max_erase_count);
294  }
295 
296  s32_t score =
297  deleted_pages_in_block * SPIFFS_GC_HEUR_W_DELET +
298  used_pages_in_block * SPIFFS_GC_HEUR_W_USED +
299  erase_age * SPIFFS_GC_HEUR_W_ERASE_AGE;
300  int cand_ix = 0;
301  SPIFFS_GC_DBG("gc_check: bix:%i del:%i use:%i score:%i\n", cur_block, deleted_pages_in_block, used_pages_in_block, score);
302  while (cand_ix < max_candidates) {
303  if (cand_blocks[cand_ix] == (spiffs_block_ix)-1) {
304  cand_blocks[cand_ix] = cur_block;
305  cand_scores[cand_ix] = score;
306  break;
307  } else if (cand_scores[cand_ix] < score) {
308  int reorder_cand_ix = max_candidates - 2;
309  while (reorder_cand_ix >= cand_ix) {
310  cand_blocks[reorder_cand_ix + 1] = cand_blocks[reorder_cand_ix];
311  cand_scores[reorder_cand_ix + 1] = cand_scores[reorder_cand_ix];
312  reorder_cand_ix--;
313  }
314  cand_blocks[cand_ix] = cur_block;
315  cand_scores[cand_ix] = score;
316  break;
317  }
318  cand_ix++;
319  }
320  (*candidate_count)++;
321  }
322 
323  cur_entry = 0;
324  cur_block++;
325  cur_block_addr += SPIFFS_CFG_LOG_BLOCK_SZ(fs);
326  } // per block
327 
328  return res;
329 }

347 // Empties given block by moving all data into free pages of another block
348 // Strategy:
349 // loop:
350 // scan object lookup for object data pages
351 // for first found id, check spix and load corresponding object index page to memory
352 // push object scan lookup entry index
353 // rescan object lookup, find data pages with same id and referenced by same object index
354 // move data page, update object index in memory
355 // when reached end of lookup, store updated object index
356 // pop object scan lookup entry index
357 // repeat loop until end of object lookup
358 // scan object lookup again for remaining object index pages, move to new page in other block
359 //
360 s32_t spiffs_gc_clean(spiffs *fs, spiffs_block_ix bix) {
361  s32_t res = SPIFFS_OK;
362  int entries_per_page = (SPIFFS_CFG_LOG_PAGE_SZ(fs) / sizeof(spiffs_obj_id));
363  int cur_entry = 0;
364  spiffs_obj_id *obj_lu_buf = (spiffs_obj_id *)fs->lu_work;
365  spiffs_gc gc;
366  spiffs_page_ix cur_pix = 0;
367  spiffs_page_object_ix_header *objix_hdr = (spiffs_page_object_ix_header *)fs->work;
368  spiffs_page_object_ix *objix = (spiffs_page_object_ix *)fs->work;
369 
370  SPIFFS_GC_DBG("gc_clean: cleaning block %i\n", bix);
371 
372  c_memset(&gc, 0, sizeof(spiffs_gc));
373  gc.state = FIND_OBJ_DATA;
374 
375  if (fs->free_cursor_block_ix == bix) {
376  // move free cursor to next block, cannot use free pages from the block we want to clean
377  fs->free_cursor_block_ix = (bix+1)%fs->block_count;
378  fs->free_cursor_obj_lu_entry = 0;
379  SPIFFS_GC_DBG("gc_clean: move free cursor to block %i\n", fs->free_cursor_block_ix);
380  }
381 
382  while (res == SPIFFS_OK && gc.state != FINISHED) {
383  SPIFFS_GC_DBG("gc_clean: state = %i entry:%i\n", gc.state, cur_entry);
384  gc.obj_id_found = 0;
385 
386  // scan through lookup pages
387  int obj_lookup_page = cur_entry / entries_per_page;
388  u8_t scan = 1;
389  // check each object lookup page
390  while (scan && res == SPIFFS_OK && obj_lookup_page < (int)SPIFFS_OBJ_LOOKUP_PAGES(fs)) {
391  int entry_offset = obj_lookup_page * entries_per_page;
392  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU | SPIFFS_OP_C_READ,
393  0, bix * SPIFFS_CFG_LOG_BLOCK_SZ(fs) + SPIFFS_PAGE_TO_PADDR(fs, obj_lookup_page),
394  SPIFFS_CFG_LOG_PAGE_SZ(fs), fs->lu_work);
395  // check each entry
396  while (scan && res == SPIFFS_OK &&
397  cur_entry - entry_offset < entries_per_page && cur_entry < (int)(SPIFFS_PAGES_PER_BLOCK(fs)-SPIFFS_OBJ_LOOKUP_PAGES(fs))) {
398  spiffs_obj_id obj_id = obj_lu_buf[cur_entry-entry_offset];
399  cur_pix = SPIFFS_OBJ_LOOKUP_ENTRY_TO_PIX(fs, bix, cur_entry);
400 
401  // act upon object id depending on gc state
402  switch (gc.state) {
403  case FIND_OBJ_DATA:
404  if (obj_id != SPIFFS_OBJ_ID_DELETED && obj_id != SPIFFS_OBJ_ID_FREE &&
405  ((obj_id & SPIFFS_OBJ_ID_IX_FLAG) == 0)) {
406  SPIFFS_GC_DBG("gc_clean: FIND_DATA state:%i - found obj id %04x\n", gc.state, obj_id);
407  gc.obj_id_found = 1;
408  gc.cur_obj_id = obj_id;
409  scan = 0;
410  }
411  break;
412  case MOVE_OBJ_DATA:
413  if (obj_id == gc.cur_obj_id) {
414  spiffs_page_header p_hdr;
415  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU2 | SPIFFS_OP_C_READ,
416  0, SPIFFS_PAGE_TO_PADDR(fs, cur_pix), sizeof(spiffs_page_header), (u8_t*)&p_hdr);
417  SPIFFS_CHECK_RES(res);
418  SPIFFS_GC_DBG("gc_clean: MOVE_DATA found data page %04x:%04x @ %04x\n", gc.cur_obj_id, p_hdr.span_ix, cur_pix);
419  if (SPIFFS_OBJ_IX_ENTRY_SPAN_IX(fs, p_hdr.span_ix) != gc.cur_objix_spix) {
420  SPIFFS_GC_DBG("gc_clean: MOVE_DATA no objix spix match, take in another run\n");
421  } else {
422  spiffs_page_ix new_data_pix;
423  if (p_hdr.flags & SPIFFS_PH_FLAG_DELET) {
424  // move page
425  res = spiffs_page_move(fs, 0, 0, obj_id, &p_hdr, cur_pix, &new_data_pix);
426  SPIFFS_GC_DBG("gc_clean: MOVE_DATA move objix %04x:%04x page %04x to %04x\n", gc.cur_obj_id, p_hdr.span_ix, cur_pix, new_data_pix);
427  SPIFFS_CHECK_RES(res);
428  // move wipes obj_lu, reload it
429  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU | SPIFFS_OP_C_READ,
430  0, bix * SPIFFS_CFG_LOG_BLOCK_SZ(fs) + SPIFFS_PAGE_TO_PADDR(fs, obj_lookup_page),
431  SPIFFS_CFG_LOG_PAGE_SZ(fs), fs->lu_work);
432  SPIFFS_CHECK_RES(res);
433  } else {
434  // page is deleted but not deleted in lookup, scrap it
435  SPIFFS_GC_DBG("gc_clean: MOVE_DATA wipe objix %04x:%04x page %04x\n", obj_id, p_hdr.span_ix, cur_pix);
436  res = spiffs_page_delete(fs, cur_pix);
437  SPIFFS_CHECK_RES(res);
438  new_data_pix = SPIFFS_OBJ_ID_FREE;
439  }
440  // update memory representation of object index page with new data page
441  if (gc.cur_objix_spix == 0) {
442  // update object index header page
443  ((spiffs_page_ix*)((u8_t *)objix_hdr + sizeof(spiffs_page_object_ix_header)))[p_hdr.span_ix] = new_data_pix;
444  SPIFFS_GC_DBG("gc_clean: MOVE_DATA wrote page %04x to objix_hdr entry %02x in mem\n", new_data_pix, SPIFFS_OBJ_IX_ENTRY(fs, p_hdr.span_ix));
445  } else {
446  // update object index page
447  ((spiffs_page_ix*)((u8_t *)objix + sizeof(spiffs_page_object_ix)))[SPIFFS_OBJ_IX_ENTRY(fs, p_hdr.span_ix)] = new_data_pix;
448  SPIFFS_GC_DBG("gc_clean: MOVE_DATA wrote page %04x to objix entry %02x in mem\n", new_data_pix, SPIFFS_OBJ_IX_ENTRY(fs, p_hdr.span_ix));
449  }
450  }
451  }
452  break;
453  case MOVE_OBJ_IX:
454  if (obj_id != SPIFFS_OBJ_ID_DELETED && obj_id != SPIFFS_OBJ_ID_FREE &&
455  (obj_id & SPIFFS_OBJ_ID_IX_FLAG)) {
456  // found an index object id
457  spiffs_page_header p_hdr;
458  spiffs_page_ix new_pix;
459  // load header
460  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU2 | SPIFFS_OP_C_READ,
461  0, SPIFFS_PAGE_TO_PADDR(fs, cur_pix), sizeof(spiffs_page_header), (u8_t*)&p_hdr);
462  SPIFFS_CHECK_RES(res);
463  if (p_hdr.flags & SPIFFS_PH_FLAG_DELET) {
464  // move page
465  res = spiffs_page_move(fs, 0, 0, obj_id, &p_hdr, cur_pix, &new_pix);
466  SPIFFS_GC_DBG("gc_clean: MOVE_OBJIX move objix %04x:%04x page %04x to %04x\n", obj_id, p_hdr.span_ix, cur_pix, new_pix);
467  SPIFFS_CHECK_RES(res);
468  spiffs_cb_object_event(fs, 0, SPIFFS_EV_IX_UPD, obj_id, p_hdr.span_ix, new_pix, 0);
469  // move wipes obj_lu, reload it
470  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU | SPIFFS_OP_C_READ,
471  0, bix * SPIFFS_CFG_LOG_BLOCK_SZ(fs) + SPIFFS_PAGE_TO_PADDR(fs, obj_lookup_page),
472  SPIFFS_CFG_LOG_PAGE_SZ(fs), fs->lu_work);
473  SPIFFS_CHECK_RES(res);
474  } else {
475  // page is deleted but not deleted in lookup, scrap it
476  SPIFFS_GC_DBG("gc_clean: MOVE_OBJIX wipe objix %04x:%04x page %04x\n", obj_id, p_hdr.span_ix, cur_pix);
477  res = spiffs_page_delete(fs, cur_pix);
478  if (res == SPIFFS_OK) {
479  spiffs_cb_object_event(fs, 0, SPIFFS_EV_IX_DEL, obj_id, p_hdr.span_ix, cur_pix, 0);
480  }
481  }
482  SPIFFS_CHECK_RES(res);
483  }
484  break;
485  default:
486  scan = 0;
487  break;
488  }
489  cur_entry++;
490  } // per entry
491  obj_lookup_page++;
492  } // per object lookup page
493 
494  if (res != SPIFFS_OK) break;
495 
496  // state finalization and switch
497  switch (gc.state) {
498  case FIND_OBJ_DATA:
499  if (gc.obj_id_found) {
500  // find out corresponding obj ix page and load it to memory
501  spiffs_page_header p_hdr;
502  spiffs_page_ix objix_pix;
503  gc.stored_scan_entry_index = cur_entry;
504  cur_entry = 0;
505  gc.state = MOVE_OBJ_DATA;
506  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU2 | SPIFFS_OP_C_READ,
507  0, SPIFFS_PAGE_TO_PADDR(fs, cur_pix), sizeof(spiffs_page_header), (u8_t*)&p_hdr);
508  SPIFFS_CHECK_RES(res);
509  gc.cur_objix_spix = SPIFFS_OBJ_IX_ENTRY_SPAN_IX(fs, p_hdr.span_ix);
510  SPIFFS_GC_DBG("gc_clean: FIND_DATA find objix span_ix:%04x\n", gc.cur_objix_spix);
511  res = spiffs_obj_lu_find_id_and_span(fs, gc.cur_obj_id | SPIFFS_OBJ_ID_IX_FLAG, gc.cur_objix_spix, 0, &objix_pix);
512  SPIFFS_CHECK_RES(res);
513  SPIFFS_GC_DBG("gc_clean: FIND_DATA found object index at page %04x\n", objix_pix);
514  res = _spiffs_rd(fs, SPIFFS_OP_T_OBJ_LU2 | SPIFFS_OP_C_READ,
515  0, SPIFFS_PAGE_TO_PADDR(fs, objix_pix), SPIFFS_CFG_LOG_PAGE_SZ(fs), fs->work);
516  SPIFFS_CHECK_RES(res);
517  SPIFFS_VALIDATE_OBJIX(objix->p_hdr, gc.cur_obj_id | SPIFFS_OBJ_ID_IX_FLAG, gc.cur_objix_spix);
518  gc.cur_objix_pix = objix_pix;
519  } else {
520  gc.state = MOVE_OBJ_IX;
521  cur_entry = 0; // restart entry scan index
522  }
523  break;
524  case MOVE_OBJ_DATA: {
525  // store modified objix (hdr) page
526  spiffs_page_ix new_objix_pix;
527  gc.state = FIND_OBJ_DATA;
528  cur_entry = gc.stored_scan_entry_index;
529  if (gc.cur_objix_spix == 0) {
530  // store object index header page
531  res = spiffs_object_update_index_hdr(fs, 0, gc.cur_obj_id | SPIFFS_OBJ_ID_IX_FLAG, gc.cur_objix_pix, fs->work, 0, 0, &new_objix_pix);
532  SPIFFS_GC_DBG("gc_clean: MOVE_DATA store modified objix_hdr page, %04x:%04x\n", new_objix_pix, 0);
533  SPIFFS_CHECK_RES(res);
534  } else {
535  // store object index page
536  res = spiffs_page_move(fs, 0, fs->work, gc.cur_obj_id | SPIFFS_OBJ_ID_IX_FLAG, 0, gc.cur_objix_pix, &new_objix_pix);
537  SPIFFS_GC_DBG("gc_clean: MOVE_DATA store modified objix page, %04x:%04x\n", new_objix_pix, objix->p_hdr.span_ix);
538  SPIFFS_CHECK_RES(res);
539  spiffs_cb_object_event(fs, 0, SPIFFS_EV_IX_UPD, gc.cur_obj_id, objix->p_hdr.span_ix, new_objix_pix, 0);
540  }
541  }
542  break;
543  case MOVE_OBJ_IX:
544  gc.state = FINISHED;
545  break;
546  default:
547  cur_entry = 0;
548  break;
549  }
550  SPIFFS_GC_DBG("gc_clean: state-> %i\n", gc.state);
551  } // while state != FINISHED
552 
553 
554  return res;
555 }
